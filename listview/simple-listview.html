Adding the UI

In your xml file add a listview object.

<RelativeLayout
    xmlns:android="http://schemas.android.com/apk/res/android"
    xmlns:tools="http://schemas.android.com/tools"
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    tools:context=".MainActivity">
    
    <ListView
        android:layout_width="match_parent"
        android:layout_height="match_parent"
        android:id="@+id/listView"
        android:layout_alignParentTop="true"
        android:layout_alignParentStart="true" />
    
</RelativeLayout>

Adding the rows

In your onCreate add rows, like:

package YOUR_PACKAGE_NAME_CHANGE_THIS;

import android.app.Activity;
import android.content.Context;
import android.support.v7.app.AppCompatActivity;
import android.os.Bundle;
import android.util.Log;
import android.view.Menu;
import android.view.MenuItem;
import android.view.View;
import android.widget.AdapterView;
import android.widget.ArrayAdapter;
import android.widget.ListView;
import android.widget.Toast;

import java.util.ArrayList;
import java.util.List;


public class MainActivity extends Activity {

    ArrayAdapter mArrayAdapter;
    ArrayList listOfThings;
    ListView listView;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        listOfThings = new ArrayList<>();
        for(int i = 0;i < 25;i++){
            listOfThings.add("Item #" + String.valueOf(i));
        }
        
        mArrayAdapter = new ArrayAdapter(this, android.R.layout.simple_list_item_1, listOfThings);
        
        listView = (ListView) findViewById(R.id.listView);
        listView.setAdapter(mArrayAdapter);
        
        listView.setOnItemClickListener(new AdapterView.OnItemClickListener() {
            @Override
            public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
                String itemValue = (String) listView.getItemAtPosition(position);
                Log.v("", "Item Value:" + itemValue);
                Toast.makeText(view.getContext(), "Item Value:" + itemValue, Toast.LENGTH_LONG).show();
            }
        }
    }
}

Hit build (play) at the top, and you should see it!
